(deftemplate enfermedad
   (slot nombre)
   (slot tratamiento)
   (slot analisis))

(defrule imprimir-diagnostico
   (enfermedad (nombre ?e) (tratamiento ?t) (analisis ?a))
   =>
   (printout t " Diagnóstico: " ?e crlf)
   (printout t " Tratamiento recomendado: " ?t crlf)
   (printout t " Análisis sugerido: " ?a crlf crlf))
